\mode<presentation>
{
    \usetheme{Frankfurt}
    \usefonttheme{professionalfonts}

    % Update the structure color theme (i.e. bullet points)
    \usecolortheme[named=UTorange]{structure}

    % Update the normal text color
    \setbeamercolor{normal text}{fg=smokeygray}
    \setbeamercolor{title}{fg=UTorange, bg=smokeygray}
    \setbeamercolor{frametitle}{fg=UTorange, bg=UTMlimestone}
    \setbeamercolor{section in head/foot}{fg=UTorange, bg=smokeygray}
    \setbeamercolor{block body}{fg=smokeygray, bg=UTMlimestone}
    \setbeamercolor{block title}{fg=UTorange, bg=smokeygray}
    
    % Prefer the flat circular bullets for all item sublevels
    \setbeamertemplate{itemize items}{{$\bullet$}}
  
    % SAY NO TO NAVIGATION SYMBOLS
    %\setbeamertemplate{navigation symbols}{}

    
    
    % add nice toggleable table of content slides to the start of each section
    \newif\ifshowtoc
    \showtoctrue % toggles to show the toc
    \AtBeginSection[]{
        \ifshowtoc % toggle this before a section via \showtoctrue or \showtocfalse
        \begin{frame}
            \frametitle{Table of Contents}
            \tableofcontents[currentsection]
        \end{frame}
        \fi
    }
    
    % i like the frame numbers
    \setbeamertemplate{footline}[frame number]{}

    % make the items gray until i click
    \setbeamercovered{transparent}
    
    % CONTROL THE COUNTING OF SLIDES: main body vs. appendix
    % use command appendix to begin the backup slides
    \usepackage{appendixnumberbeamer}
    \newcommand{\beginbackup}{
       \newcounter{framenumbervorappendix}
       \setcounter{framenumbervorappendix}{\value{framenumber}}
       
       \setbeamertemplate{footline}
       {
         \leavevmode%
         \hline
         box{%
           \begin{beamercolorbox}[wd=\paperwidth,ht=2.25ex,dp=1ex,right]{footlinecolor}%
    %         \insertframenumber  %comment \hspace*{2ex} 
           \end{beamercolorbox}}%
         \vskip0pt%
       }
     }
     \newcommand{\backupend}{
       \addtocounter{framenumbervorappendix}{-\value{framenumber}}
       \addtocounter{framenumber}{\value{framenumbervorappendix}} 
    }

}
